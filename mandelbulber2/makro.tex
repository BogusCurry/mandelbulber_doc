% Makro simple image with 75% width
% #1 : file path
% #2 : caption
% #3 : label
\def\simpleImageWithCaptionFullWidth#1#2#3{%
	\begin{figure}[H]
		\begin{center}
			\includegraphics[width=1.00\linewidth]{#1}
			\vspace*{-6mm}
			\caption{#2}
			\vspace*{-8mm}
			\label{#3}
		\end{center}
	\end{figure}
}

% Makro two images with two captions
% #1 : file path 1
% #2 : caption 1
% #3 : label 1
% #4 : file path 2
% #5 : caption 2
% #6 : label 2
\def\twoImagesWithTwoCaptionsFullWidth#1#2#3#4#5#6{%
	\begin{figure}[H]
		
		\begin{center}
			\begin{minipage}[b]{0.4\textwidth}
				\begin{center}
					\includegraphics[height=0.7\linewidth]{#1}
					\vspace*{-2mm}
					\caption{#2}
					\label{#3}
				\end{center}
			\end{minipage}
			\begin{minipage}[b]{0.4\textwidth}

					\caption{#5}
					\label{#6}
				\end{center}
			\end{minipage}
		\end{center}
	\end{figure}
}
